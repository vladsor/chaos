00033 /* Calling environment, plus possibly a saved signal mask.  */
00034 typedef struct __jmp_buf_tag    /* C++ doesn't like tagless structs.  */
00035   {
00036     /* NOTE: The machine-dependent definitions of `__sigsetjmp'
00037        assume that a `jmp_buf' begins with a `__jmp_buf' and that
00038        `__mask_was_saved' follows it.  Do not move these members
00039        or add others before it.  */
00040     __jmp_buf __jmpbuf;     /* Calling environment.  */
00041     int __mask_was_saved;   /* Saved the signal mask?  */
00042     __sigset_t __saved_mask;    /* Saved signal mask.  */
00043   } jmp_buf[1];

00050 /* Store the calling environment in ENV, not saving the signal mask.

00054 /* Store the calling environment in ENV, also saving the

00057 extern int __sigsetjmp (jmp_buf __env, int __savemask) __THROW;

00059 #ifndef __FAVOR_BSD
00060 /* Do not save the signal mask.  This is equivalent to the `_setjmp'
00061    BSD function.  */
00062 # define setjmp(env)    _setjmp (env)
00063 #else
00064 /* We are in 4.3 BSD-compatibility mode in which `setjmp'
00065    saves the signal mask like `sigsetjmp (ENV, 1)'.  We have to
00066    define a macro since ISO C says `setjmp' is one.  */
00067 # define setjmp(env)    setjmp (env)
00068 #endif /* Favor BSD.  */

00071 /* Jump to the environment saved in ENV, making the

00073 extern void longjmp (jmp_buf __env, int __val)

00076 /* Same.  Usually `_longjmp' is used with `_setjmp', which does not save
