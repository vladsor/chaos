00009 /* ioctl command encoding: 32 bits total, command in lower 16 bits,
00010  * size of the parameter structure in the lower 14 bits of the
00011  * upper 16 bits.
00012  * Encoding the size of the parameter structure in the ioctl request
00013  * is useful for catching programs compiled with old versions
00014  * and to avoid overwriting user space outside the user buffer area.
00015  * The highest 2 bits are reserved for indicating the ``access mode''.
00016  * NOTE: This limits the max parameter size to 16kB -1 !
00017  */
00018 
00019 /*
00020  * The following is for compatibility across the various Linux
00021  * platforms.  The i386 ioctl numbering scheme doesn't really enforce
00022  * a type field.  De facto, however, the top 8 bits of the lower 16
00023  * bits are indeed used as a type field, so we might just as well make
00024  * this explicit here.  Please be sure to use the decoding macros
00025  * below from now on.
00026  */
00027 #define _IOC_NRBITS 8
00028 #define _IOC_TYPEBITS   8
00029 #define _IOC_SIZEBITS   14
00030 #define _IOC_DIRBITS    2
00031 
00032 #define _IOC_NRMASK ((1 << _IOC_NRBITS)-1)
00033 #define _IOC_TYPEMASK   ((1 << _IOC_TYPEBITS)-1)
00034 #define _IOC_SIZEMASK   ((1 << _IOC_SIZEBITS)-1)
00035 #define _IOC_DIRMASK    ((1 << _IOC_DIRBITS)-1)
00036 
00037 #define _IOC_NRSHIFT    0
00038 #define _IOC_TYPESHIFT  (_IOC_NRSHIFT+_IOC_NRBITS)
00039 #define _IOC_SIZESHIFT  (_IOC_TYPESHIFT+_IOC_TYPEBITS)
00040 #define _IOC_DIRSHIFT   (_IOC_SIZESHIFT+_IOC_SIZEBITS)
00041 
00042 /*
00043  * Direction bits.
00044  */
00045 #define _IOC_NONE   0U
00046 #define _IOC_WRITE  1U
00047 #define _IOC_READ   2U
00048 
00049 #define _IOC(dir,type,nr,size) \
00050     (((dir)  << _IOC_DIRSHIFT) | \
00051      ((type) << _IOC_TYPESHIFT) | \
00052      ((nr)   << _IOC_NRSHIFT) | \
00053      ((size) << _IOC_SIZESHIFT))
00054 
00055 /* used to create numbers */
00056 #define _IO(type,nr)        _IOC(_IOC_NONE,(type),(nr),0)
00057 #define _IOR(type,nr,size)  _IOC(_IOC_READ,(type),(nr),sizeof(size))
00058 #define _IOW(type,nr,size)  _IOC(_IOC_WRITE,(type),(nr),sizeof(size))
00059 #define _IOWR(type,nr,size) _IOC(_IOC_READ|_IOC_WRITE,(type),(nr),sizeof(size))
00060 
00061 /* used to decode ioctl numbers.. */
00062 #define _IOC_DIR(nr)        (((nr) >> _IOC_DIRSHIFT) & _IOC_DIRMASK)
00063 #define _IOC_TYPE(nr)       (((nr) >> _IOC_TYPESHIFT) & _IOC_TYPEMASK)
00064 #define _IOC_NR(nr)     (((nr) >> _IOC_NRSHIFT) & _IOC_NRMASK)
00065 #define _IOC_SIZE(nr)       (((nr) >> _IOC_SIZESHIFT) & _IOC_SIZEMASK)
00066 
00067 /* ...and for the drivers/sound files... */
00068 
00069 #define IOC_IN      (_IOC_WRITE << _IOC_DIRSHIFT)
00070 #define IOC_OUT     (_IOC_READ << _IOC_DIRSHIFT)
00071 #define IOC_INOUT   ((_IOC_WRITE|_IOC_READ) << _IOC_DIRSHIFT)
00072 #define IOCSIZE_MASK    (_IOC_SIZEMASK << _IOC_SIZESHIFT)
00073 #define IOCSIZE_SHIFT   (_IOC_SIZESHIFT)
