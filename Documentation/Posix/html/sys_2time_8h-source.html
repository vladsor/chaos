00142 /* Convenience macros for operations on timevals.
00143    NOTE: `timercmp' does not work for >= or <=.  */
00144 # define timerisset(tvp)    ((tvp)->tv_sec || (tvp)->tv_usec)
00145 # define timerclear(tvp)    ((tvp)->tv_sec = (tvp)->tv_usec = 0)
00146 # define timercmp(a, b, CMP)                              \
00147   (((a)->tv_sec == (b)->tv_sec) ?                         \
00148    ((a)->tv_usec CMP (b)->tv_usec) :                          \
00149    ((a)->tv_sec CMP (b)->tv_sec))
00150 # define timeradd(a, b, result)                           \
00151   do {                                        \
00152     (result)->tv_sec = (a)->tv_sec + (b)->tv_sec;                 \
00153     (result)->tv_usec = (a)->tv_usec + (b)->tv_usec;                  \
00154     if ((result)->tv_usec >= 1000000)                         \
00155       {                                       \
00156     ++(result)->tv_sec;                           \
00157     (result)->tv_usec -= 1000000;                         \
00158       }                                       \
00159   } while (0)

00161   do {                                        \

00164     if ((result)->tv_usec < 0) {                          \
