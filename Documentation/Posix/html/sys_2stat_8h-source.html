00282 /* To allow the `struct stat' structure and the file type `mode_t'
00283    bits to vary without changing shared library major version number,
00284    the `stat' family of functions and `mknod' are in fact inline
00285    wrappers around calls to `xstat', `fxstat', `lxstat', and `xmknod',
00286    which all take a leading version-number argument designating the
00287    data structure and bits used.  <bits/stat.h> defines _STAT_VER with
00288    the version number corresponding to `struct stat' as defined in
00289    that file; and _MKNOD_VER with the version number corresponding to
00290    the S_IF* macros defined therein.  It is arranged that when not
00291    inlined these function are always statically linked; that way a
00292    dynamically-linked executable always encodes the version number
00293    corresponding to the data structures it uses, so the `x' functions
00294    in the shared library can adapt without needing to recompile all
00295    callers.  */
00296 
00297 #ifndef _STAT_VER
00298 # define _STAT_VER  0
00299 #endif
00300 #ifndef _MKNOD_VER
00301 # define _MKNOD_VER 0
00302 #endif
