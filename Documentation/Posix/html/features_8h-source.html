00022 /* These are defined by the user (or the compiler)
00023    to specify the desired environment:
00024 
00025    __STRICT_ANSI__  ISO Standard C.
00026    _ISOC99_SOURCE   Extensions to ISO C89 from ISO C99.
00027    _POSIX_SOURCE    IEEE Std 1003.1.
00028    _POSIX_C_SOURCE  If ==1, like _POSIX_SOURCE; if >=2 add IEEE Std 1003.2;
00029             if >=199309L, add IEEE Std 1003.1b-1993;
00030             if >=199506L, add IEEE Std 1003.1c-1995
00031    _XOPEN_SOURCE    Includes POSIX and XPG things.  Set to 500 if
00032             Single Unix conformance is wanted, to 600 for the
00033             upcoming sixth revision.
00034    _XOPEN_SOURCE_EXTENDED XPG things and X/Open Unix extensions.
00035    _LARGEFILE_SOURCE    Some more functions for correct standard I/O.
00036    _LARGEFILE64_SOURCE  Additional functionality from LFS for large files.
00037    _FILE_OFFSET_BITS=N  Select default filesystem interface.
00038    _BSD_SOURCE      ISO C, POSIX, and 4.3BSD things.
00039    _SVID_SOURCE     ISO C, POSIX, and SVID things.
00040    _GNU_SOURCE      All of the above, plus GNU extensions.
00041    _REENTRANT       Select additionally reentrant object.
00042    _THREAD_SAFE     Same as _REENTRANT, often used by other systems.
00043 
00044    The `-ansi' switch to the GNU C compiler defines __STRICT_ANSI__.
00045    If none of these are defined, the default is to have _SVID_SOURCE,
00046    _BSD_SOURCE, and _POSIX_SOURCE set to one and _POSIX_C_SOURCE set to
00047    199506L.  If more than one of these are defined, they accumulate.
00048    For example __STRICT_ANSI__, _POSIX_SOURCE and _POSIX_C_SOURCE
00049    together give you ISO C, 1003.1, and 1003.2, but nothing else.
00050 
00051    These are defined by this file and are used by the
00052    header files to decide what to declare or define:
00053 
00054    __USE_ISOC99     Define ISO C99 things.
00055    __USE_POSIX      Define IEEE Std 1003.1 things.
00056    __USE_POSIX2     Define IEEE Std 1003.2 things.
00057    __USE_POSIX199309    Define IEEE Std 1003.1, and .1b things.
00058    __USE_POSIX199506    Define IEEE Std 1003.1, .1b, .1c and .1i things.
00059    __USE_XOPEN      Define XPG things.
00060    __USE_XOPEN_EXTENDED Define X/Open Unix things.
00061    __USE_UNIX98     Define Single Unix V2 things.
00062    __USE_XOPEN2K        Define XPG6 things.
00063    __USE_LARGEFILE  Define correct standard I/O things.
00064    __USE_LARGEFILE64    Define LFS things with separate names.
00065    __USE_FILE_OFFSET64  Define 64bit interface as default.
00066    __USE_BSD        Define 4.3BSD things.
00067    __USE_SVID       Define SVID things.
00068    __USE_MISC       Define things common to BSD and System V Unix.
00069    __USE_GNU        Define GNU extensions.
00070    __USE_REENTRANT  Define reentrant/thread-safe *_r functions.
00071    __FAVOR_BSD      Favor 4.3BSD things in cases of conflict.
00072 
00073    The macros `__GNU_LIBRARY__', `__GLIBC__', and `__GLIBC_MINOR__' are
00074    defined by this file unconditionally.  `__GNU_LIBRARY__' is provided
00075    only for compatibility.  All new code should use the other symbols
00076    to test for features.
00077 
00078    All macros listed above as possibly being defined by this file are
00079    explicitly undefined if they are not explicitly defined.
00080    Feature-test macros that are not defined by the user or compiler
00081    but are implied by the other feature-test macros defined (or by the
00082    lack of any definitions) are defined by the file.  */

00263 /* Convenience macros to test the versions of glibc and gcc.
00264    Use them like this:
00265    #if __GNUC_PREREQ (2,8)
00266    ... code requiring gcc 2.8 or later ...
00267    #endif
00268    Note - they won't work for gcc1 or glibc1, since the _MINOR macros
00269    were not defined then.  */
00270 #if defined __GNUC__ && defined __GNUC_MINOR__
00271 # define __GNUC_PREREQ(maj, min) \
00272     ((__GNUC__ << 16) + __GNUC_MINOR__ >= ((maj) << 16) + (min))
00273 #else
00274 # define __GNUC_PREREQ(maj, min) 0
00275 #endif
