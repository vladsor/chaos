00049 /* Return a mask that includes the bit for SIG only.  */
00050 # define __sigmask(sig) \
00051   (((unsigned long int) 1) << (((sig) - 1) % (8 * sizeof (unsigned long int))))

00053 /* Return the word index for SIG.  */
00054 # define __sigword(sig) (((sig) - 1) / (8 * sizeof (unsigned long int)))
