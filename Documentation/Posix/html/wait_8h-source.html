00119 /* Wait for a child matching PID to die.
00120    If PID is greater than 0, match any process whose process ID is PID.
00121    If PID is (pid_t) -1, match any process.
00122    If PID is (pid_t) 0, match any process with the
00123    same process group as the current process.
00124    If PID is less than -1, match any process whose
00125    process group is the absolute value of PID.
00126    If the WNOHANG bit is set in OPTIONS, and that child
00127    is not already dead, return (pid_t) 0.  If successful,
00128    return PID and store the dead child's status in STAT_LOC.
00129    Return (pid_t) -1 for errors.  If the WUNTRACED bit is
00130    set in OPTIONS, return status for stopped children; otherwise don't.  */
00131 extern __pid_t waitpid __P ((__pid_t __pid, int *__stat_loc,
00132                  int __options));
