00028 /* If WIFEXITED(STATUS), the low-order 8 bits of the status.  */
00029 #define __WEXITSTATUS(status)   (((status) & 0xff00) >> 8)
00030 
00031 /* If WIFSIGNALED(STATUS), the terminating signal.  */
00032 #define __WTERMSIG(status)  ((status) & 0x7f)
00033 
00034 /* If WIFSTOPPED(STATUS), the signal that stopped the child.  */
00035 #define __WSTOPSIG(status)  __WEXITSTATUS(status)
00036 
00037 /* Nonzero if STATUS indicates normal termination.  */
00038 #define __WIFEXITED(status) (__WTERMSIG(status) == 0)
00039 
00040 /* Nonzero if STATUS indicates termination by a signal.  */
00041 #ifdef  __GNUC__
00042 # define __WIFSIGNALED(status) \
00043   (__extension__ ({ int __status = (status);                      \
00044             !__WIFSTOPPED(__status) && !__WIFEXITED(__status); }))
00045 #else   /* Not GCC.  */
00046 # define __WIFSIGNALED(status)  (!__WIFSTOPPED(status) && !__WIFEXITED(status))
00047 #endif  /* GCC.  */

00049 /* Nonzero if STATUS indicates the child is stopped.  */
00050 #define __WIFSTOPPED(status)    (((status) & 0xff) == 0x7f)

00052 /* Nonzero if STATUS indicates the child dumped core.  */
00053 #define __WCOREDUMP(status) ((status) & __WCOREFLAG)
