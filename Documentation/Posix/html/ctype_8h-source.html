00031 #ifndef _ISbit
00032 /* These are all the characteristics of characters.
00033    If there get to be more than 16 distinct characteristics,
00034    many things must be changed that use `unsigned short int's.
00035 
00036    The characteristics are stored always in network byte order (big
00037    endian).  We define the bit value interpretations here dependent on the
00038    machine's byte order.  */
00039 
00040 # include <endian.h>
00041 # if __BYTE_ORDER == __BIG_ENDIAN
00042 #  define _ISbit(bit)   (1 << (bit))
00043 # else /* __BYTE_ORDER == __LITTLE_ENDIAN */
00044 #  define _ISbit(bit)   ((bit) < 8 ? ((1 << (bit)) << 8) : ((1 << (bit)) >> 8))
00045 # endif
00046 
00047 enum
00048 {
00049   _ISupper = _ISbit (0),    /* UPPERCASE.  */
00050   _ISlower = _ISbit (1),    /* lowercase.  */
00051   _ISalpha = _ISbit (2),    /* Alphabetic.  */
00052   _ISdigit = _ISbit (3),    /* Numeric.  */
00053   _ISxdigit = _ISbit (4),   /* Hexadecimal numeric.  */
00054   _ISspace = _ISbit (5),    /* Whitespace.  */
00055   _ISprint = _ISbit (6),    /* Printing.  */
00056   _ISgraph = _ISbit (7),    /* Graphical.  */
00057   _ISblank = _ISbit (8),    /* Blank (usually SPC and TAB).  */
00058   _IScntrl = _ISbit (9),    /* Control character.  */
00059   _ISpunct = _ISbit (10),   /* Punctuation.  */
00060   _ISalnum = _ISbit (11)    /* Alphanumeric.  */
00061 };
00062 #endif /* ! _ISbit  */

00064 /* These are defined in ctype-info.c.
00065    The declarations here must match those in localeinfo.h.
00066 
00067    These point into arrays of 384, so they can be indexed by any `unsigned
00068    char' value [0,255]; by EOF (-1); or by any `signed char' value
00069    [-128,-1).  ISO C requires that the ctype functions work for `unsigned
00070    char' values and for EOF; we also support negative `signed char' values
00071    for broken old programs.  The case conversion arrays are of `int's
00072    rather than `unsigned char's because tolower (EOF) must be EOF, which
00073    doesn't fit into an `unsigned char'.  But today more important is that
00074    the arrays are also used for multi-byte character sets.  */
00075 extern __const unsigned short int *__ctype_b;   /* Characteristics.  */
00076 extern __const __int32_t *__ctype_tolower; /* Case conversions.  */
00077 extern __const __int32_t *__ctype_toupper; /* Case conversions.  */

00131 /* This code is needed for the optimized mapping functions.  */
00132 #define __tobody(c, f, a, args) \
00133   (__extension__                                  \
00134    ({ int __res;                                  \
00135       if (sizeof (c) > 1)                             \
00136     {                                     \
00137       if (__builtin_constant_p (c))                       \
00138         {                                     \
00139           int __c = (c);                              \
00140           __res = __c < -128 || __c > 255 ? __c : a[__c];             \
00141         }                                     \
00142       else                                    \
00143         __res = f args;                           \
00144     }                                     \
00145       else                                    \
00146     __res = a[(int) (c)];                             \
00147       __res; }))
