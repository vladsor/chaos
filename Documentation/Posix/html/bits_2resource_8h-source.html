00152 /* Structure which says how much of each resource has been used.  */
00153 struct rusage
00154   {
00155     /* Total amount of user time used.  */
00156     struct timeval ru_utime;
00157     /* Total amount of system time used.  */
00158     struct timeval ru_stime;
00159     /* Maximum resident set size (in kilobytes).  */
00160     long int ru_maxrss;
00161     /* Amount of sharing of text segment memory
00162        with other processes (kilobyte-seconds).  */
00163     long int ru_ixrss;
00164     /* Amount of data segment memory used (kilobyte-seconds).  */
00165     long int ru_idrss;
00166     /* Amount of stack memory used (kilobyte-seconds).  */
00167     long int ru_isrss;
00168     /* Number of soft page faults (i.e. those serviced by reclaiming
00169        a page from the list of pages awaiting reallocation.  */
00170     long int ru_minflt;
00171     /* Number of hard page faults (i.e. those that required I/O).  */
00172     long int ru_majflt;
00173     /* Number of times a process was swapped out of physical memory.  */
00174     long int ru_nswap;
00175     /* Number of input operations via the file system.  Note: This
00176        and `ru_oublock' do not include operations with the cache.  */
00177     long int ru_inblock;
00178     /* Number of output operations via the file system.  */
00179     long int ru_oublock;
00180     /* Number of IPC messages sent.  */
00181     long int ru_msgsnd;
00182     /* Number of IPC messages received.  */
00183     long int ru_msgrcv;
00184     /* Number of signals delivered.  */
00185     long int ru_nsignals;
00186     /* Number of voluntary context switches, i.e. because the process
00187        gave up the process before it had to (usually to wait for some
00188        resource to be available).  */
00189     long int ru_nvcsw;
00190     /* Number of involuntary context switches, i.e. a higher priority process
00191        became runnable or the current process used up its time slice.  */
00192     long int ru_nivcsw;
00193   };
