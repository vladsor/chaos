00281 /* Structure passed to `sigvec'.  */

00285     int sv_mask;        /* Mask of signals to be blocked.  */

00291 /* Bits in `sv_flags'.  */
00292 # define SV_ONSTACK (1 << 0)/* Take the signal on the signal stack.  */
00293 # define SV_INTERRUPT   (1 << 1)/* Do not restart system calls.  */
00294 # define SV_RESETHAND   (1 << 2)/* Reset handler to SIG_DFL on receipt.  */

00297 /* If VEC is non-NULL, set the handler for SIG to the `sv_handler' member
00298    of VEC.  The signals in `sv_mask' will be blocked while the handler runs.
00299    If the SV_RESETHAND bit is set in `sv_flags', the handler for SIG will be
00300    reset to SIG_DFL before `sv_handler' is entered.  If OVEC is non-NULL,
00301    it is filled in with the old information for SIG.  */
00302 extern int sigvec (int __sig, __const struct sigvec *__vec,
