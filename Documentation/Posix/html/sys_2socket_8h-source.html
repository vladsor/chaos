00059 /* This is the type we use for generic socket address arguments.
00060 
00061    With GCC 2.7 and later, the funky union causes redeclarations or

00063    G++ 2.7 does not support transparent unions so there we want the
00064    old-style declaration, too.  */
00065 #if defined __cplusplus || !__GNUC_PREREQ (2, 7) || !defined __USE_GNU
00066 # define __SOCKADDR_ARG     struct sockaddr *__restrict
00067 # define __CONST_SOCKADDR_ARG   __const struct sockaddr *

00074   __SOCKADDR_ONETYPE (sockaddr_ax25) \
00075   __SOCKADDR_ONETYPE (sockaddr_dl) \

00086 # define __SOCKADDR_ONETYPE(type) struct type *__restrict __##type##__;
